96  template <typename U>
97  iterator insert_func(U&& value)
98  {
99  std::vector<skip_list_node*> update(MAX_LEVEL, nullptr);
100  skip_list_node* current = head_;
101 
102  for (int i = MAX_LEVEL - 1; i >= 0; --i)
103  {
104  while (current->next[i] != tail_ && comp_(current->next[i]->value.value(), value))
105  {
106  current = current->next[i];
107  }
108  update[i] = current;
109  }
110 
111  int new_level = random_level();
112  skip_list_node* new_node = node_allocator_traits::allocate(alloc_, 1);
113  node_allocator_traits::construct(alloc_, new_node, std::forward<U>(value), new_level);
114 
115  for (int i = 0; i < new_level; ++i)
116  {
117  new_node->next[i] = update[i]->next[i];
118  update[i]->next[i] = new_node;
119  }
120 
121  new_node->prev = update[0];
122  if (new_node->next[0])
123  {
124  new_node->next[0]->prev = new_node;
125  }
126  ++size_;
127  return iterator(new_node);
128  }
